你这个需求，建议核心思想定成一句话：
模拟主流程只负责跑 1000 次；事件只旁路观察；指标只在结果容器里收集，不反向影响模拟。
也就是：
SimulationRunner
    负责跑流程

ISimulationObserver
    负责监听事件

SimulationResult
    负责收集结果

MetricSink / MetricBag
    负责收集不同指标

不要让“事件”和“指标”污染主流程。

1. 总体结构
你要的是这个结构：
Simulate(1000)
    for i in 0..999
        RunOneSimulation(context)
            行为组前事件
            执行行为组
                move 前事件
                move
                move 后事件
            行为组后事件

            成功事件 / 失败事件

        写入结果容器

主流程应该长这样：
public sealed class SimulationRunner
{
    private readonly ISimulationPolicy _policy;
    private readonly ISimulationObserver _observer;

    public SimulationRunner(
        ISimulationPolicy policy,
        ISimulationObserver observer)
    {
        _policy = policy;
        _observer = observer;
    }

    public SimulationBatchResult Simulate(
        LevelCore level,
        int count)
    {
        var result = new SimulationBatchResult(count);

        for (int i = 0; i < count; i++)
        {
            var runResult = RunOne(level, i);
            result.Add(runResult);
        }

        return result;
    }

    private SimulationRunResult RunOne(
        LevelCore originLevel,
        int simulationIndex)
    {
        var level = originLevel.CloneForSimulation();

        var context = new SimulationContext(
            simulationIndex,
            level);

        using var timing = SimulationTiming.Start();

        while (!level.IsFinished)
        {
            var group = _policy.NextBehaviourGroup(level);

            if (group.IsEmpty)
                break;

            SafeNotifyBeforeBehaviourGroup(context, group);

            foreach (var move in group.Moves)
            {
                SafeNotifyBeforeMove(context, move);

                level.ApplyMove(move);

                SafeNotifyAfterMove(context, move);
            }

            SafeNotifyAfterBehaviourGroup(context, group);
        }

        var success = level.IsSuccess;

        if (success)
            SafeNotifySuccess(context);
        else
            SafeNotifyFailure(context);

        return new SimulationRunResult(
            simulationIndex,
            success,
            level.MoveCount,
            timing.ElapsedMilliseconds);
    }

    private void SafeNotifyBeforeMove(SimulationContext context, Move move)
    {
        try { _observer.OnBeforeMove(context, move); }
        catch { }
    }

    private void SafeNotifyAfterMove(SimulationContext context, Move move)
    {
        try { _observer.OnAfterMove(context, move); }
        catch { }
    }

    private void SafeNotifyBeforeBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group)
    {
        try { _observer.OnBeforeBehaviourGroup(context, group); }
        catch { }
    }

    private void SafeNotifyAfterBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group)
    {
        try { _observer.OnAfterBehaviourGroup(context, group); }
        catch { }
    }

    private void SafeNotifySuccess(SimulationContext context)
    {
        try { _observer.OnSuccess(context); }
        catch { }
    }

    private void SafeNotifyFailure(SimulationContext context)
    {
        try { _observer.OnFailure(context); }
        catch { }
    }
}

这里有一个重点：
try { observer.OnXXX(...); }
catch { }

这就是你说的：
事件不影响主流程。
即使指标统计炸了、日志炸了、观察器炸了，模拟本身继续跑。

2. 事件接口设计
事件不要做成 C# 的 event Action<...>，那样容易有委托分配、组合复杂、异常传播问题。
建议直接做接口：
public interface ISimulationObserver
{
    void OnBeforeBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group);

    void OnAfterBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group);

    void OnBeforeMove(
        SimulationContext context,
        Move move);

    void OnAfterMove(
        SimulationContext context,
        Move move);

    void OnSuccess(SimulationContext context);

    void OnFailure(SimulationContext context);
}

然后给一个空实现：
public sealed class EmptySimulationObserver : ISimulationObserver
{
    public static readonly EmptySimulationObserver Instance = new();

    private EmptySimulationObserver() { }

    public void OnBeforeBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group) { }

    public void OnAfterBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group) { }

    public void OnBeforeMove(
        SimulationContext context,
        Move move) { }

    public void OnAfterMove(
        SimulationContext context,
        Move move) { }

    public void OnSuccess(SimulationContext context) { }

    public void OnFailure(SimulationContext context) { }
}

这样主流程不需要判断 null：
_observer.OnBeforeMove(context, move);


3. 多个观察器怎么办？
不要让主流程知道多个 observer。
做一个组合观察器：
public sealed class CompositeSimulationObserver : ISimulationObserver
{
    private readonly ISimulationObserver[] _observers;

    public CompositeSimulationObserver(ISimulationObserver[] observers)
    {
        _observers = observers;
    }

    public void OnBeforeBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group)
    {
        foreach (var observer in _observers)
            observer.OnBeforeBehaviourGroup(context, group);
    }

    public void OnAfterBehaviourGroup(
        SimulationContext context,
        BehaviourGroup group)
    {
        foreach (var observer in _observers)
            observer.OnAfterBehaviourGroup(context, group);
    }

    public void OnBeforeMove(
        SimulationContext context,
        Move move)
    {
        foreach (var observer in _observers)
            observer.OnBeforeMove(context, move);
    }

    public void OnAfterMove(
        SimulationContext context,
        Move move)
    {
        foreach (var observer in _observers)
            observer.OnAfterMove(context, move);
    }

    public void OnSuccess(SimulationContext context)
    {
        foreach (var observer in _observers)
            observer.OnSuccess(context);
    }

    public void OnFailure(SimulationContext context)
    {
        foreach (var observer in _observers)
            observer.OnFailure(context);
    }
}

如果你想保证每个 observer 都不能影响主流程，可以把 try/catch 放在这里，而不是 Runner 里：
private static void SafeInvoke(Action action)
{
    try { action(); }
    catch { }
}

不过为了性能，不建议用 Action 包装。可以在每个方法里直接写 try/catch。

可以这样：

    public readonly LevelCore Level;

    public SimulationContext(

        LevelCore level)

        SimulationIndex = simulationIndex;

如果你要暴露更多运行时信息：
public readonly struct SimulationContext
{
    public readonly int SimulationIndex;
    public readonly LevelCore Level;
    public readonly long StartTimestamp;

    public SimulationContext(
        int simulationIndex,
        LevelCore level,
        long startTimestamp)
    {
        SimulationIndex = simulationIndex;
        Level = level;
        StartTimestamp = startTimestamp;
    }
}

然后提供一个时间工具：
public static class SimulationClock
{
    public static long NowTicks()
    {
        return Stopwatch.GetTimestamp();
    }

    public static double GetElapsedMilliseconds(long startTimestamp)
    {
        long end = Stopwatch.GetTimestamp();
        long delta = end - startTimestamp;

        return delta * 1000.0 / Stopwatch.Frequency;
    }
}

这样事件里可以拿时间，但不需要影响主流程。

5. 行为组：直接 Move
你说：
行为组，直接 move行为组前 / 行为组后
那 BehaviourGroup 可以非常薄：
public readonly struct BehaviourGroup
{
    public readonly ReadOnlyMemory<Move> Moves;

    public bool IsEmpty => Moves.Length == 0;

    public BehaviourGroup(ReadOnlyMemory<Move> moves)
    {
        Moves = moves;
    }
}

如果你追求热路径低分配，后面可以改成：
public readonly ref struct BehaviourGroupSpan
{
    public readonly ReadOnlySpan<Move> Moves;

    public bool IsEmpty => Moves.Length == 0;

    public BehaviourGroupSpan(ReadOnlySpan<Move> moves)
    {
        Moves = moves;
    }
}

但第一版不建议马上上 ref struct，否则接口、observer、存储都会受限制。
第一版可以先用：
Move[] buffer
int count

例如：
public readonly struct BehaviourGroup
{
    public readonly Move[] Moves;
    public readonly int Count;

    public bool IsEmpty => Count == 0;

    public BehaviourGroup(Move[] moves, int count)
    {
        Moves = moves;
        Count = count;
    }
}

执行时：
for (int i = 0; i < group.Count; i++)
{
    var move = group.Moves[i];

    observer.OnBeforeMove(context, move);
    level.ApplyMove(move);
    observer.OnAfterMove(context, move);
}

这比一开始搞复杂枚举器更直观。

6. 外部接口怎么定义
你需要把“模拟策略”和“模拟框架”分开。
模拟框架负责流程：
public sealed class SimulationRunner

策略负责决定下一组行为：
public interface ISimulationPolicy
{
    BehaviourGroup NextBehaviourGroup(LevelCore level);
}

外部使用：
var policy = new RandomSimulationPolicy();
var observer = new MetricsSimulationObserver();

var runner = new SimulationRunner(policy, observer);

SimulationBatchResult result = runner.Simulate(level, 1000);

这就是你说的：
模拟外部接口。
外部只关心：
runner.Simulate(level, 1000)

内部事件、指标、时间都不暴露成主流程复杂度。

7. 结果容器：建议分两层
你现在头疼的是：
结果容器如何定义：用于收集不同指标的值。
这个不要一开始就做成：
Dictionary<string, object>

那样最灵活，但类型差、性能差、后面会乱。
建议分两层：
SimulationBatchResult
    固定核心结果

MetricBag
    扩展指标


7.1 固定核心结果
先定义稳定的核心指标：
public sealed class SimulationBatchResult
{
    private readonly SimulationRunResult[] _runs;
    private int _count;

    public int Count => _count;

    public int SuccessCount { get; private set; }
    public int FailureCount => _count - SuccessCount;

    public double SuccessRate =>
        _count == 0 ? 0 : (double)SuccessCount / _count;

    public double FailRate =>
        _count == 0 ? 0 : (double)FailureCount / _count;

    public MetricBag Metrics { get; } = new();

    public SimulationBatchResult(int capacity)
    {
        _runs = new SimulationRunResult[capacity];
    }

    public void Add(SimulationRunResult result)
    {
        _runs[_count++] = result;

        if (result.Success)
            SuccessCount++;
    }

    public ReadOnlySpan<SimulationRunResult> Runs =>
        _runs.AsSpan(0, _count);
}

单次结果：
public readonly struct SimulationRunResult
{
    public readonly int SimulationIndex;
    public readonly bool Success;
    public readonly int MoveCount;
    public readonly double ElapsedMilliseconds;

    public SimulationRunResult(
        int simulationIndex,
        bool success,
        int moveCount,
        double elapsedMilliseconds)
    {
        SimulationIndex = simulationIndex;
        Success = success;
        MoveCount = moveCount;
        ElapsedMilliseconds = elapsedMilliseconds;
    }
}

这一层放稳定字段：
SimulationIndex
Success
MoveCount
ElapsedMilliseconds

这些是模拟框架天然会产生的，不需要通过扩展指标绕一圈。

7.2 扩展指标：MetricBag
扩展指标用 MetricBag：
public sealed class MetricBag
{
    private readonly Dictionary<string, double> _doubleMetrics = new();
    private readonly Dictionary<string, int> _intMetrics = new();

    public void SetDouble(string name, double value)
    {
        _doubleMetrics[name] = value;
    }

    public void AddDouble(string name, double value)
    {
        if (_doubleMetrics.TryGetValue(name, out var old))
            _doubleMetrics[name] = old + value;
        else
            _doubleMetrics[name] = value;
    }

    public void SetInt(string name, int value)
    {
        _intMetrics[name] = value;
    }

    public void AddInt(string name, int value)
    {
        if (_intMetrics.TryGetValue(name, out var old))
            _intMetrics[name] = old + value;
        else
            _intMetrics[name] = value;
    }

    public bool TryGetDouble(string name, out double value)
    {
        return _doubleMetrics.TryGetValue(name, out value);
    }

    public bool TryGetInt(string name, out int value)
    {
        return _intMetrics.TryGetValue(name, out value);
    }
}

第一版这样就够了。
不要一开始支持所有类型：
object?
string
bool
array
histogram

指标大部分最后都是：
int
double

10. 结果容器不要一次做成万能系统
你真正需要的是三个层级：
第一层：单次结果 SimulationRunResult
第二层：批量结果 SimulationBatchResult
第三层：扩展指标 MetricBag

结构如下：
public sealed class SimulationBatchResult
{
    public int Count { get; }
    public int SuccessCount { get; }
    public int FailureCount { get; }
    public double FailRate { get; }

    public ReadOnlySpan<SimulationRunResult> Runs { get; }

    public MetricBag Metrics { get; }
}

其中：
Count / SuccessCount / FailureCount / FailRate

是框架稳定字段。
Metrics

是扩展字段。
这样你不会纠结：
所有指标都放哪里？
答案是：
稳定核心指标：直接做属性
实验性指标：放 MetricBag


11. 更适合你项目的版本
结合你之前的代码风格，我建议第一版定成这样：
public sealed class SimulationRunner
{
    public SimulationBatchResult Simulate(
        LevelCore level,
        int simulationCount,
        ISimulationPolicy policy,
        ISimulationObserver? observer = null)
    {
        observer ??= EmptySimulationObserver.Instance;

        var result = new SimulationBatchResult(simulationCount);

        for (int i = 0; i < simulationCount; i++)
        {
            var runResult = RunOne(level, i, policy, observer);
            result.Add(runResult);
        }

        return result;
    }

    private SimulationRunResult RunOne(
        LevelCore originLevel,
        int simulationIndex,
        ISimulationPolicy policy,
        ISimulationObserver observer)
    {
        var level = originLevel.CloneForSimulation();

        long start = Stopwatch.GetTimestamp();

        var context = new SimulationContext(
            simulationIndex,
            level,
            start);

        while (!level.IsFinished)
        {
            BehaviourGroup group = policy.NextBehaviourGroup(level);

            if (group.IsEmpty)
                break;

            SafeBeforeBehaviourGroup(observer, context, group);

            for (int i = 0; i < group.Count; i++)
            {
                Move move = group.Moves[i];

                SafeBeforeMove(observer, context, move);

                level.ApplyMove(move);

                SafeAfterMove(observer, context, move);
            }

            SafeAfterBehaviourGroup(observer, context, group);
        }

        bool success = level.IsSuccess;

        if (success)
            SafeSuccess(observer, context);
        else
            SafeFailure(observer, context);

        double elapsedMs =
            (Stopwatch.GetTimestamp() - start) * 1000.0 / Stopwatch.Frequency;

        return new SimulationRunResult(
            simulationIndex,
            success,
            level.MoveCount,
            elapsedMs);
    }

    private static void SafeBeforeMove(
        ISimulationObserver observer,
        SimulationContext context,
        Move move)
    {
        try { observer.OnBeforeMove(context, move); }
        catch { }
    }

    private static void SafeAfterMove(
        ISimulationObserver observer,
        SimulationContext context,
        Move move)
    {
        try { observer.OnAfterMove(context, move); }
        catch { }
    }

    private static void SafeBeforeBehaviourGroup(
        ISimulationObserver observer,
        SimulationContext context,
        BehaviourGroup group)
    {
        try { observer.OnBeforeBehaviourGroup(context, group); }
        catch { }
    }

    private static void SafeAfterBehaviourGroup(
        ISimulationObserver observer,
        SimulationContext context,
        BehaviourGroup group)
    {
        try { observer.OnAfterBehaviourGroup(context, group); }
        catch { }
    }

    private static void SafeSuccess(
        ISimulationObserver observer,
        SimulationContext context)
    {
        try { observer.OnSuccess(context); }
        catch { }
    }

    private static void SafeFailure(
        ISimulationObserver observer,
        SimulationContext context)
    {
        try { observer.OnFailure(context); }
        catch { }
    }
}

配套：

12. 最终建议
你的模拟框架第一版可以这样定：
SimulationRunner
    跑 1000 次

ISimulationPolicy
    决定行为组

BehaviourGroup
    包含一组 Move

ISimulationObserver
    监听事件，不影响主流程

SimulationBatchResult
    收集稳定结果

MetricBag
    收集扩展指标

主路径保持非常干净：
for 1000 次
    行为组前
    move 前
    move
    move 后
    行为组后
    成功 / 失败

最重要的设计判断是：
事件系统不是业务流程的一部分，而是旁路观察系统。结果容器不是万能对象，而是“稳定字段 + 扩展指标袋”。
